ionic에서 css를 사용한 flip card 구현하기

다음과 같이 간략하게 과정을 정리해 본다.

1) html

  
toggle
  
<section class="container">
      
<div id="card">
          
<figure class="front">1</figure>
          
<figure class="back">2</figure>
      
</div>
  
</section>

2) css

  
.container {
      
width: 200px;
      
height: 260px;
      
position: relative;
      
perspective: 800px;
  
}

#card {
      
width: 100%;
      
height: 100%;
      
position: absolute;
      
transform-style: preserve-3d;
      
transition: transform 1s;
  
}

#card figure {
      
display: block;
      
position: absolute;
      
width: 100%;
      
height: 100%;
      
backface-visibility: hidden;
  
}

#card .front {
      
background: red;
  
}
  
#card .back {
      
background: blue;
      
transform: rotateY( 180deg );
  
}

#card.flipped {
      
transform: rotateY( 180deg );
  
}
